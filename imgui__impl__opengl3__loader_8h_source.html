8//
9// YOU SHOULD NOT NEED TO INCLUDE/USE THIS DIRECTLY. THIS IS USED BY imgui_impl_opengl3.cpp ONLY.
10// THE REST OF YOUR APP SHOULD USE A DIFFERENT GL LOADER: ANY GL LOADER OF YOUR CHOICE.
11//
12// IF YOU GET BUILD ERRORS IN THIS FILE (commonly macro redefinitions or function redefinitions):
13// IT LIKELY MEANS THAT YOU ARE BUILDING 'imgui_impl_opengl3.cpp' OR INCLUDING 'imgui_impl_opengl3_loader.h'
14// IN THE SAME COMPILATION UNIT AS ONE OF YOUR FILE WHICH IS USING A THIRD-PARTY OPENGL LOADER.
15// (e.g. COULD HAPPEN IF YOU ARE DOING A UNITY/JUMBO BUILD, OR INCLUDING .CPP FILES FROM OTHERS)
16// YOU SHOULD NOT BUILD BOTH IN THE SAME COMPILATION UNIT.
17// BUT IF YOU REALLY WANT TO, you can '#define IMGUI_IMPL_OPENGL_LOADER_CUSTOM' and imgui_impl_opengl3.cpp
18// WILL NOT BE USING OUR LOADER, AND INSTEAD EXPECT ANOTHER/YOUR LOADER TO BE AVAILABLE IN THE COMPILATION UNIT.
19//
20// Regenerate with:
21// python3 gl3w_gen.py --output ../imgui/backends/imgui_impl_opengl3_loader.h --ref ../imgui/backends/imgui_impl_opengl3.cpp ./extra_symbols.txt

725static int open_libs(void)
726{
727 // On Linux we have two APIs to get process addresses: EGL and GLX.
728 // EGL is supported under both X11 and Wayland, whereas GLX is X11-specific.
729
730 libgl = NULL;
731 libegl = NULL;
732 libglx = NULL;
733
734 // First check what's already loaded, the windowing library might have
735 // already loaded either EGL or GLX and we want to use the same one.
736
737 if (is_library_loaded("libEGL.so.1", &libegl) ||
738 is_library_loaded("libGLX.so.0", &libglx)) {
739 libgl = dlopen("libOpenGL.so.0", RTLD_LAZY | RTLD_LOCAL);
740 if (libgl)
741 return GL3W_OK;
742 else
743 close_libgl();
744 }
745
746 if (is_library_loaded("libGL.so", &libgl))
747 return GL3W_OK;
748 if (is_library_loaded("libGL.so.1", &libgl))
749 return GL3W_OK;
750 if (is_library_loaded("libGL.so.3", &libgl))
751 return GL3W_OK;
752
753 // Neither is already loaded, so we have to load one. Try EGL first
754 // because it is supported under both X11 and Wayland.
755
756 // Load OpenGL + EGL
757 libgl = dlopen("libOpenGL.so.0", RTLD_LAZY | RTLD_LOCAL);
758 libegl = dlopen("libEGL.so.1", RTLD_LAZY | RTLD_LOCAL);
759 if (libgl && libegl)
760 return GL3W_OK;
761 else
762 close_libgl();
763
764 // Fall back to legacy libGL, which includes GLX
765 // While most systems use libGL.so.1, NetBSD seems to use that libGL.so.3. See https://github.com/ocornut/imgui/issues/6983
766 libgl = dlopen("libGL.so", RTLD_LAZY | RTLD_LOCAL);
767 if (!libgl)
768 libgl = dlopen("libGL.so.1", RTLD_LAZY | RTLD_LOCAL);
769 if (!libgl)
770 libgl = dlopen("libGL.so.3", RTLD_LAZY | RTLD_LOCAL);
771
772 if (libgl)
773 return GL3W_OK;
774
775 return GL3W_ERROR_LIBRARY_OPEN;
776}

778static int open_libgl(void)
779{
780 int res = open_libs();
781 if (res)
782 return res;
783
784 if (libegl)

786 else if (libglx)

788 else

792 close_libgl();

794 }

796 return GL3W_OK;

799static GL3WglProc get_proc(const char* proc)
800{
801 GL3WglProc res = NULL;
802
803 // Before EGL version 1.5, eglGetProcAddress doesn't support querying core
804 // functions and may return a dummy function if we try, so try to load the
805 // function from the GL library directly first.
806 if (libegl)
807 *(void**)(&res) = dlsym(libgl, proc);
808
809 if (!res)
810 res = gl_get_proc_address(proc);
811
812 if (!libegl && !res)
813 *(void**)(&res) = dlsym(libgl, proc);
814
815 return res;
816}
